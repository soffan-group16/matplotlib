10def check_shared(axs, x_shared, y_shared): 

11 """ 

12 x_shared and y_shared are n x n boolean matrices; entry (i, j) indicates 

13 whether the x (or y) axes of subplots i and j should be shared. 

14 """ 

15 for (i1, ax1), (i2, ax2), (i3, (name, shared)) in itertools.product( 

16 enumerate(axs), 

17 enumerate(axs), 

18 enumerate(zip("xy", [x_shared, y_shared]))): 

19 if i2 <= i1: 

20 continue 

21 assert \ 

22 (getattr(axs[0], "_shared_{}_axes".format(name)).joined(ax1, ax2) 

23 == shared[i1, i2]), \ 

24 "axes %i and %i incorrectly %ssharing %s axis" % ( 

25 i1, i2, "not " if shared[i1, i2] else "", name)
