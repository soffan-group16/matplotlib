90def common_texification(text): 

91 r""" 

92 Do some necessary and/or useful substitutions for texts to be included in 

93 LaTeX documents. 

94 

95 This distinguishes text-mode and math-mode by replacing the math separator 

96 ``$`` with ``\(\displaystyle %s\)``. Escaped math separators (``\$``) 

97 are ignored. 

98 

99 The following characters are escaped in text segments: ``_^$%`` 

100 """ 

101 # Sometimes, matplotlib adds the unknown command \mathdefault. 

102 # Not using \mathnormal instead since this looks odd for the latex cm font. 

103 text = _replace_mathdefault(text) 

104 text = text.replace("\N{MINUS SIGN}", r"\ensuremath{-}") 

105 # split text into normaltext and inline math parts 

106 parts = re_mathsep.split(text) 

107 for i, s in enumerate(parts): 

108 if not i % 2: 

109 # textmode replacements 

110 s = _replace_escapetext(s) 

111 else: 

112 # mathmode replacements 

113 s = r"\(\displaystyle %s\)" % s 

114 parts[i] = s 

115 return "".join(parts)

327 def get_width_height_descent(self, text, prop): 

328 """ 

329 Get the width, total height and descent for a text typeset by the 

330 current LaTeX environment. 

331 """ 

332

334 prop_cmds = _font_properties_str(prop) 

335 textbox = "\\sbox0{%s %s}" % (prop_cmds, text) 

336 

337 # check cache

339 return self.str_cache[textbox]

342 self._stdin_writeln(textbox)

345 except LatexError as e:

350 self._stdin_writeln(r"\typeout{\the\wd0,\the\ht0,\the\dp0}") 

351 # read answer from latex and advance to the next prompt 

352 try: 

353 answer = self._expect_prompt() 

354 except LatexError as e: 

355 raise ValueError("Error processing '{}'\nLaTeX Output:\n{}" 

356 .format(text, e.latex_output)) from e 

357 

358 # parse metrics from the answer string 

359 try: 

360 width, height, offset = answer.splitlines()[0].split(",") 

361 except Exception as err: 

362 raise ValueError("Error processing '{}'\nLaTeX Output:\n{}" 

363 .format(text, answer)) from err 

364 w, h, o = float(width[:-2]), float(height[:-2]), float(offset[:-2]) 

365 

366 # the height returned from LaTeX goes from base to top. 

367 # the height matplotlib expects goes from bottom to top. 

368 self.str_cache[textbox] = (w, h + o, o)
