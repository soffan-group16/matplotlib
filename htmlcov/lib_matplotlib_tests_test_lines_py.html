29# Runtimes on a loaded system are inherently flaky. Not so much that a rerun 

30# won't help, hopefully. 

31@pytest.mark.flaky(reruns=3) 

32def test_invisible_Line_rendering(): 

33 """ 

34 GitHub issue #1256 identified a bug in Line.draw method 

35 

36 Despite visibility attribute set to False, the draw method was not 

37 returning early enough and some pre-rendering code was executed 

38 though not necessary. 

39 

40 Consequence was an excessive draw time for invisible Line instances 

41 holding a large number of points (Npts> 10**6) 

42 """ 

43 # Creates big x and y data: 

44 N = 10**7 

45 x = np.linspace(0, 1, N) 

46 y = np.random.normal(size=N) 

47 

48 # Create a plot figure: 

49 fig = plt.figure() 

50 ax = plt.subplot() 

51 

52 # Create a "big" Line instance: 

53 l = mlines.Line2D(x, y) 

54 l.set_visible(False) 

55 # but don't add it to the Axis instance `ax` 

56 

57 # [here Interactive panning and zooming is pretty responsive] 

58 # Time the canvas drawing: 

59 t_no_line = min(timeit.repeat(fig.canvas.draw, number=1, repeat=3)) 

60 # (gives about 25 ms) 

61 

62 # Add the big invisible Line: 

63 ax.add_line(l) 

64 

65 # [Now interactive panning and zooming is very slow] 

66 # Time the canvas drawing: 

67 t_invisible_line = min(timeit.repeat(fig.canvas.draw, number=1, repeat=3)) 

68 # gives about 290 ms for N = 10**7 pts 

69 

70 slowdown_factor = t_invisible_line / t_no_line 

71 slowdown_threshold = 2 # trying to avoid false positive failures 

72 assert slowdown_factor < slowdown_threshold
