1""" 

2Adjust subplot layouts so that there are no overlapping axes or axes 

3decorations. All axes decorations are dealt with (labels, ticks, titles, 

4ticklabels) and some dependent artists are also dealt with (colorbar, 

5suptitle). 

6 

7Layout is done via `~matplotlib.gridspec`, with one constraint per gridspec, 

8so it is possible to have overlapping axes if the gridspecs overlap (i.e. 

9using `~matplotlib.gridspec.GridSpecFromSubplotSpec`). Axes placed using 

10``figure.subplots()`` or ``figure.add_subplots()`` will participate in the 

11layout. Axes manually placed via ``figure.add_axes()`` will not. 

12 

13See Tutorial: :doc:`/tutorials/intermediate/constrainedlayout_guide` 

14"""

25""" 

26General idea: 

27------------- 

28 

29First, a figure has a gridspec that divides the figure into nrows and ncols, 

30with heights and widths set by ``height_ratios`` and ``width_ratios``, 

31often just set to 1 for an equal grid. 

32 

33Subplotspecs that are derived from this gridspec can contain either a 

34``SubPanel``, a ``GridSpecFromSubplotSpec``, or an axes. The ``SubPanel`` and 

35``GridSpecFromSubplotSpec`` are dealt with recursively and each contain an 

36analogous layout. 

37 

38Each ``GridSpec`` has a ``_layoutgrid`` attached to it. The ``_layoutgrid`` 

39has the same logical layout as the ``GridSpec``. Each row of the grid spec 

40has a top and bottom "margin" and each column has a left and right "margin". 

41The "inner" height of each row is constrained to be the same (or as modified 

42by ``height_ratio``), and the "inner" width of each column is 

43constrained to be the same (as modified by ``width_ratio``), where "inner" 

44is the width or height of each column/row minus the size of the margins. 

45 

46Then the size of the margins for each row and column are determined as the 

47max width of the decorators on each axes that has decorators in that margin. 

48For instance, a normal axes would have a left margin that includes the 

49left ticklabels, and the ylabel if it exists. The right margin may include a 

50colorbar, the bottom margin the xaxis decorations, and the top margin the 

51title. 

52 

53With these constraints, the solver then finds appropriate bounds for the 

54columns and rows. It's possible that the margins take up the whole figure, 

55in which case the algorithm is not applied and a warning is raised. 

56 

57See the tutorial doc:`/tutorials/intermediate/constrainedlayout_guide` 

58for more discussion of the algorithm with examples. 

59"""

63def do_constrained_layout(fig, renderer, h_pad, w_pad, 

64 hspace=None, wspace=None): 

65 """ 

66 Do the constrained_layout. Called at draw time in 

67 ``figure.constrained_layout()`` 

68 

69 Parameters 

70 ---------- 

71 fig : Figure 

72 ``Figure`` instance to do the layout in. 

73 

74 renderer : Renderer 

75 Renderer to use. 

76 

77 h_pad, w_pad : float 

78 Padding around the axes elements in figure-normalized units. 

79 

80 hspace, wspace : float 

81 Fraction of the figure to dedicate to space between the 

82 axes. These are evenly spread between the gaps between the axes. 

83 A value of 0.2 for a three-column layout would have a space 

84 of 0.1 of the figure width between each column. 

85 If h/wspace < h/w_pad, then the pads are used instead. 

86 """ 

87 

88 # list of unique gridspecs that contain child axes: 

89 gss = set() 

90 for ax in fig.axes: 

91 if hasattr(ax, 'get_subplotspec'): 

92 gs = ax.get_subplotspec().get_gridspec() 

93 if gs._layoutgrid is not None: 

94 gss.add(gs) 

95 gss = list(gss) 

96 if len(gss) == 0: 

97 _api.warn_external('There are no gridspecs with layoutgrids. ' 

98 'Possibly did not call parent GridSpec with the' 

99 ' "figure" keyword') 

100 

101 for _ in range(2): 

102 # do the algorithm twice. This has to be done because decorations 

103 # change size after the first re-position (i.e. x/yticklabels get 

104 # larger/smaller). This second reposition tends to be much milder, 

105 # so doing twice makes things work OK. 

106 

107 # make margins for all the axes and subfigures in the 

108 # figure. Add margins for colorbars... 

109 _make_layout_margins(fig, renderer, h_pad=h_pad, w_pad=w_pad, 

110 hspace=hspace, wspace=wspace) 

111 _make_margin_suptitles(fig, renderer, h_pad=h_pad, w_pad=w_pad) 

112 

113 # if a layout is such that a columns (or rows) margin has no 

114 # constraints, we need to make all such instances in the grid 

115 # match in margin size. 

116 _match_submerged_margins(fig) 

117 

118 # update all the variables in the layout. 

119 fig._layoutgrid.update_variables() 

120 

121 if _check_no_collapsed_axes(fig): 121 ↛ 125line 121 didn't jump to line 125, because the condition on line 121 was never false

122 _reposition_axes(fig, renderer, h_pad=h_pad, w_pad=w_pad, 

123 hspace=hspace, wspace=wspace) 

124 else: 

125 _api.warn_external('constrained_layout not applied because ' 

126 'axes sizes collapsed to zero. Try making ' 

127 'figure larger or axes decorations smaller.') 

128 _reset_margins(fig)

153def _get_margin_from_padding(object, *, w_pad=0, h_pad=0, 

154 hspace=0, wspace=0): 

155 

156 ss = object._subplotspec 

157 gs = ss.get_gridspec() 

158 lg = gs._layoutgrid 

159 

160 if hasattr(gs, 'hspace'): 

161 _hspace = (gs.hspace if gs.hspace is not None else hspace) 

162 _wspace = (gs.wspace if gs.wspace is not None else wspace) 

163 else: 

164 _hspace = (gs._hspace if gs._hspace is not None else hspace) 

165 _wspace = (gs._wspace if gs._wspace is not None else wspace) 

166 

167 _wspace = _wspace / 2 

168 _hspace = _hspace / 2 

169 

170 nrows, ncols = gs.get_geometry() 

171 # there are two margins for each direction. The "cb" 

172 # margins are for pads and colorbars, the non-"cb" are 

173 # for the axes decorations (labels etc). 

174 margin = {'leftcb': w_pad, 'rightcb': w_pad, 

175 'bottomcb': h_pad, 'topcb': h_pad, 

176 'left': 0, 'right': 0, 

177 'top': 0, 'bottom': 0} 

178 if _wspace / ncols > w_pad: 

179 if ss.colspan.start > 0: 

180 margin['leftcb'] = _wspace / ncols 

181 if ss.colspan.stop < ncols: 

182 margin['rightcb'] = _wspace / ncols 

183 if _hspace / nrows > h_pad: 

184 if ss.rowspan.stop < nrows: 

185 margin['bottomcb'] = _hspace / nrows 

186 if ss.rowspan.start > 0: 

187 margin['topcb'] = _hspace / nrows 

188 

189 return margin

192def _make_layout_margins(fig, renderer, *, w_pad=0, h_pad=0, 

193 hspace=0, wspace=0): 

194 """ 

195 For each axes, make a margin between the *pos* layoutbox and the 

196 *axes* layoutbox be a minimum size that can accommodate the 

197 decorations on the axis. 

198 

199 Then make room for colorbars. 

200 """ 

201 for panel in fig.subfigs: # recursively make child panel margins 

202 ss = panel._subplotspec 

203 _make_layout_margins(panel, renderer, w_pad=w_pad, h_pad=h_pad, 

204 hspace=hspace, wspace=wspace) 

205 

206 margins = _get_margin_from_padding(panel, w_pad=0, h_pad=0, 

207 hspace=hspace, wspace=wspace) 

208 panel._layoutgrid.parent.edit_outer_margin_mins(margins, ss) 

209 

210 for ax in fig._localaxes.as_list(): 

211 if not hasattr(ax, 'get_subplotspec') or not ax.get_in_layout(): 

212 continue 

213 

214 ss = ax.get_subplotspec() 

215 gs = ss.get_gridspec() 

216 nrows, ncols = gs.get_geometry() 

217 

218 if gs._layoutgrid is None: 

219 return 

220 

221 margin = _get_margin_from_padding(ax, w_pad=w_pad, h_pad=h_pad, 

222 hspace=hspace, wspace=wspace) 

223 margin0 = margin.copy() 

224 pos, bbox = _get_pos_and_bbox(ax, renderer) 

225 # the margin is the distance between the bounding box of the axes 

226 # and its position (plus the padding from above) 

227 margin['left'] += pos.x0 - bbox.x0 

228 margin['right'] += bbox.x1 - pos.x1 

229 # remember that rows are ordered from top: 

230 margin['bottom'] += pos.y0 - bbox.y0 

231 margin['top'] += bbox.y1 - pos.y1 

232 

233 # make margin for colorbars. These margins go in the 

234 # padding margin, versus the margin for axes decorators. 

235 for cbax in ax._colorbars: 

236 # note pad is a fraction of the parent width... 

237 pad = _colorbar_get_pad(cbax) 

238 # colorbars can be child of more than one subplot spec: 

239 cbp_rspan, cbp_cspan = _get_cb_parent_spans(cbax) 

240 loc = cbax._colorbar_info['location'] 

241 cbpos, cbbbox = _get_pos_and_bbox(cbax, renderer) 

242 if loc == 'right': 

243 if cbp_cspan.stop == ss.colspan.stop: 

244 # only increase if the colorbar is on the right edge 

245 margin['rightcb'] += cbbbox.width + pad 

246 elif loc == 'left': 

247 if cbp_cspan.start == ss.colspan.start: 247 ↛ 258line 247 didn't jump to line 258, because the condition on line 247 was never false

248 # only increase if the colorbar is on the left edge 

249 margin['leftcb'] += cbbbox.width + pad 

250 elif loc == 'top': 

251 if cbp_rspan.start == ss.rowspan.start: 251 ↛ 258line 251 didn't jump to line 258, because the condition on line 251 was never false

252 margin['topcb'] += cbbbox.height + pad 

253 else: 

254 if cbp_rspan.stop == ss.rowspan.stop: 

255 margin['bottomcb'] += cbbbox.height + pad 

256 # If the colorbars are wider than the parent box in the 

257 # cross direction 

258 if loc in ['top', 'bottom']: 

259 if (cbp_cspan.start == ss.colspan.start and 

260 cbbbox.x0 < bbox.x0): 

261 margin['left'] += bbox.x0 - cbbbox.x0 

262 if (cbp_cspan.stop == ss.colspan.stop and 

263 cbbbox.x1 > bbox.x1): 

264 margin['right'] += cbbbox.x1 - bbox.x1 

265 # or taller: 

266 if loc in ['left', 'right']: 

267 if (cbp_rspan.stop == ss.rowspan.stop and 

268 cbbbox.y0 < bbox.y0): 

269 margin['bottom'] += bbox.y0 - cbbbox.y0 

270 if (cbp_rspan.start == ss.rowspan.start and 

271 cbbbox.y1 > bbox.y1): 

272 margin['top'] += cbbbox.y1 - bbox.y1 

273 # pass the new margins down to the layout grid for the solution... 

274 gs._layoutgrid.edit_outer_margin_mins(margin, ss)

311def _match_submerged_margins(fig): 

312 """ 

313 Make the margins that are submerged inside an Axes the same size. 

314 

315 This allows axes that span two columns (or rows) that are offset 

316 from one another to have the same size. 

317 

318 This gives the proper layout for something like:: 

319 fig = plt.figure(constrained_layout=True) 

320 axs = fig.subplot_mosaic("AAAB\nCCDD") 

321 

322 Without this routine, the axes D will be wider than C, because the 

323 margin width between the two columns in C has no width by default, 

324 whereas the margins between the two columns of D are set by the 

325 width of the margin between A and B. However, obviously the user would 

326 like C and D to be the same size, so we need to add constraints to these 

327 "submerged" margins. 

328 

329 This routine makes all the interior margins the same, and the spacing 

330 between the three columns in A and the two column in C are all set to the 

331 margins between the two columns of D. 

332 

333 See test_constrained_layout::test_constrained_layout12 for an example. 

334 """ 

335 

336 for panel in fig.subfigs: 

337 _match_submerged_margins(panel) 

338 

339 axs = [a for a in fig.get_axes() if (hasattr(a, 'get_subplotspec') 

340 and a.get_in_layout())] 

341 

342 for ax1 in axs: 

343 ss1 = ax1.get_subplotspec() 

344 lg1 = ss1.get_gridspec()._layoutgrid 

345 if lg1 is None: 

346 axs.remove(ax1) 

347 continue 

348 

349 # interior columns: 

350 if len(ss1.colspan) > 1: 

351 maxsubl = np.max( 

352 lg1.margin_vals['left'][ss1.colspan[1:]] + 

353 lg1.margin_vals['leftcb'][ss1.colspan[1:]] 

354 ) 

355 maxsubr = np.max( 

356 lg1.margin_vals['right'][ss1.colspan[:-1]] + 

357 lg1.margin_vals['rightcb'][ss1.colspan[:-1]] 

358 ) 

359 for ax2 in axs: 

360 ss2 = ax2.get_subplotspec() 

361 lg2 = ss2.get_gridspec()._layoutgrid 

362 if lg2 is not None and len(ss2.colspan) > 1: 

363 maxsubl2 = np.max( 

364 lg2.margin_vals['left'][ss2.colspan[1:]] + 

365 lg2.margin_vals['leftcb'][ss2.colspan[1:]]) 

366 if maxsubl2 > maxsubl: 

367 maxsubl = maxsubl2 

368 maxsubr2 = np.max( 

369 lg2.margin_vals['right'][ss2.colspan[:-1]] + 

370 lg2.margin_vals['rightcb'][ss2.colspan[:-1]]) 

371 if maxsubr2 > maxsubr: 

372 maxsubr = maxsubr2 

373 for i in ss1.colspan[1:]: 

374 lg1.edit_margin_min('left', maxsubl, cell=i) 

375 for i in ss1.colspan[:-1]: 

376 lg1.edit_margin_min('right', maxsubr, cell=i) 

377 

378 # interior rows: 

379 if len(ss1.rowspan) > 1: 

380 maxsubt = np.max( 

381 lg1.margin_vals['top'][ss1.rowspan[1:]] + 

382 lg1.margin_vals['topcb'][ss1.rowspan[1:]] 

383 ) 

384 maxsubb = np.max( 

385 lg1.margin_vals['bottom'][ss1.rowspan[:-1]] + 

386 lg1.margin_vals['bottomcb'][ss1.rowspan[:-1]] 

387 ) 

388 

389 for ax2 in axs: 

390 ss2 = ax2.get_subplotspec() 

391 lg2 = ss2.get_gridspec()._layoutgrid 

392 if lg2 is not None: 392 ↛ 389line 392 didn't jump to line 389, because the condition on line 392 was never false

393 if len(ss2.rowspan) > 1: 

394 maxsubt = np.max([np.max( 

395 lg2.margin_vals['top'][ss2.rowspan[1:]] + 

396 lg2.margin_vals['topcb'][ss2.rowspan[1:]] 

397 ), maxsubt]) 

398 maxsubb = np.max([np.max( 

399 lg2.margin_vals['bottom'][ss2.rowspan[:-1]] + 

400 lg2.margin_vals['bottomcb'][ss2.rowspan[:-1]] 

401 ), maxsubb]) 

402 for i in ss1.rowspan[1:]: 

403 lg1.edit_margin_min('top', maxsubt, cell=i) 

404 for i in ss1.rowspan[:-1]: 

405 lg1.edit_margin_min('bottom', maxsubb, cell=i)

408def _get_cb_parent_spans(cbax): 

409 """ 

410 Figure out which subplotspecs this colorbar belongs to: 

411 """ 

412 rowstart = np.inf 

413 rowstop = -np.inf 

414 colstart = np.inf 

415 colstop = -np.inf 

416 for parent in cbax._colorbar_info['parents']: 

417 ss = parent.get_subplotspec() 

418 rowstart = min(ss.rowspan.start, rowstart) 

419 rowstop = max(ss.rowspan.stop, rowstop) 

420 colstart = min(ss.colspan.start, colstart) 

421 colstop = max(ss.colspan.stop, colstop) 

422 

423 rowspan = range(rowstart, rowstop) 

424 colspan = range(colstart, colstop) 

425 return rowspan, colspan

505def _reposition_colorbar(cbax, renderer, *, offset=None): 

506 """ 

507 Place the colorbar in its new place. 

508 

509 Parameters 

510 ---------- 

511 cbax : Axes 

512 Axes for the colorbar 

513 

514 renderer : 

515 w_pad, h_pad : float 

516 width and height padding (in fraction of figure) 

517 hspace, wspace : float 

518 width and height padding as fraction of figure size divided by 

519 number of columns or rows 

520 margin : array-like 

521 offset the colorbar needs to be pushed to in order to 

522 account for multiple colorbars 

523 """ 

524 

525 parents = cbax._colorbar_info['parents'] 

526 gs = parents[0].get_gridspec() 

527 ncols, nrows = gs.ncols, gs.nrows 

528 fig = cbax.figure 

529 trans_fig_to_subfig = fig.transFigure - fig.transSubfigure 

530 

531 cb_rspans, cb_cspans = _get_cb_parent_spans(cbax) 

532 bboxparent = gs._layoutgrid.get_bbox_for_cb(rows=cb_rspans, cols=cb_cspans) 

533 pb = gs._layoutgrid.get_inner_bbox(rows=cb_rspans, cols=cb_cspans) 

534 

535 location = cbax._colorbar_info['location'] 

536 anchor = cbax._colorbar_info['anchor'] 

537 fraction = cbax._colorbar_info['fraction'] 

538 aspect = cbax._colorbar_info['aspect'] 

539 shrink = cbax._colorbar_info['shrink'] 

540 

541 cbpos, cbbbox = _get_pos_and_bbox(cbax, renderer) 

542 

543 # Colorbar gets put at extreme edge of outer bbox of the subplotspec 

544 # It needs to be moved in by: 1) a pad 2) its "margin" 3) by 

545 # any colorbars already added at this location: 

546 cbpad = _colorbar_get_pad(cbax) 

547 if location in ('left', 'right'): 

548 # fraction and shrink are fractions of parent 

549 pbcb = pb.shrunk(fraction, shrink).anchored(anchor, pb) 

550 # The colorbar is at the left side of the parent. Need 

551 # to translate to right (or left) 

552 if location == 'right': 

553 lmargin = cbpos.x0 - cbbbox.x0 

554 dx = bboxparent.x1 - pbcb.x0 + offset['right'] 

555 dx += cbpad + lmargin 

556 offset['right'] += cbbbox.width + cbpad 

557 pbcb = pbcb.translated(dx, 0) 

558 else: 

559 lmargin = cbpos.x0 - cbbbox.x0 

560 dx = bboxparent.x0 - pbcb.x0 # edge of parent 

561 dx += -cbbbox.width - cbpad + lmargin - offset['left'] 

562 offset['left'] += cbbbox.width + cbpad 

563 pbcb = pbcb.translated(dx, 0) 

564 else: # horizontal axes: 

565 pbcb = pb.shrunk(shrink, fraction).anchored(anchor, pb) 

566 if location == 'top': 

567 bmargin = cbpos.y0 - cbbbox.y0 

568 dy = bboxparent.y1 - pbcb.y0 + offset['top'] 

569 dy += cbpad + bmargin 

570 offset['top'] += cbbbox.height + cbpad 

571 pbcb = pbcb.translated(0, dy) 

572 else: 

573 bmargin = cbpos.y0 - cbbbox.y0 

574 dy = bboxparent.y0 - pbcb.y0 

575 dy += -cbbbox.height - cbpad + bmargin - offset['bottom'] 

576 offset['bottom'] += cbbbox.height + cbpad 

577 pbcb = pbcb.translated(0, dy) 

578 

579 pbcb = trans_fig_to_subfig.transform_bbox(pbcb) 

580 cbax.set_transform(fig.transSubfigure) 

581 cbax._set_position(pbcb) 

582 cbax.set_aspect(aspect, anchor=anchor, adjustable='box') 

583 return offset

586def _reset_margins(fig): 

587 """ 

588 Reset the margins in the layoutboxes of fig. 

589 

590 Margins are usually set as a minimum, so if the figure gets smaller 

591 the minimum needs to be zero in order for it to grow again. 

592 """ 

593 for span in fig.subfigs: 

594 _reset_margins(span) 

595 for ax in fig.axes: 

596 if hasattr(ax, 'get_subplotspec') and ax.get_in_layout(): 

597 ss = ax.get_subplotspec() 

598 gs = ss.get_gridspec() 

599 if gs._layoutgrid is not None: 

600 gs._layoutgrid.reset_margins() 

601 fig._layoutgrid.reset_margins()

604def _colorbar_get_pad(cax): 

605 parents = cax._colorbar_info['parents'] 

606 gs = parents[0].get_gridspec() 

607 

608 cb_rspans, cb_cspans = _get_cb_parent_spans(cax) 

609 bboxouter = gs._layoutgrid.get_inner_bbox(rows=cb_rspans, cols=cb_cspans) 

610
